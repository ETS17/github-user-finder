# github_user_finder.py
import tkinter as tk
from tkinter import ttk, messagebox
import requests
import json
import os

class GitHubUserFinder:
    def __init__(self, root):
        self.root = root
        self.root.title("🐙 GitHub User Finder")
        self.root.geometry("700x550")
        self.root.resizable(False, False)
        self.root.configure(bg="#2d2d2d")
        
        # Файл для сохранения избранных
        self.favourites_file = "favourites.json"
        self.favourites = self.load_favourites()
        
        # Текущий список результатов поиска
        self.search_results = []
        self.current_display_favourites = False
        
        self.create_widgets()
    
    def load_favourites(self):
        """Загрузка избранных пользователей из JSON"""
        if os.path.exists(self.favourites_file):
            try:
                with open(self.favourites_file, "r", encoding="utf-8") as f:
                    return json.load(f)
            except:
                return []
        return []
    
    def save_favourites(self):
        """Сохранение избранных в JSON"""
        with open(self.favourites_file, "w", encoding="utf-8") as f:
            json.dump(self.favourites, f, ensure_ascii=False, indent=2)
    
    def create_widgets(self):
        # Заголовок
        title = tk.Label(
            self.root,
            text="🐙 GitHub User Finder",
            font=("Arial", 18, "bold"),
            bg="#2d2d2d",
            fg="white"
        )
        title.pack(pady=15)
        
        # Поле поиска
        search_frame = tk.Frame(self.root, bg="#2d2d2d")
        search_frame.pack(pady=10)
        
        tk.Label(
            search_frame,
            text="Введите имя пользователя:",
            font=("Arial", 11),
            bg="#2d2d2d",
            fg="#cccccc"
        ).pack(side=tk.LEFT, padx=5)
        
        self.search_entry = tk.Entry(
            search_frame,
            width=30,
            font=("Arial", 11),
            relief="solid",
            bd=1
        )
        self.search_entry.pack(side=tk.LEFT, padx=5)
        self.search_entry.bind("<Return>", lambda event: self.search_users())
        
        # Кнопки управления
        button_frame = tk.Frame(self.root, bg="#2d2d2d")
        button_frame.pack(pady=10)
        
        tk.Button(
            button_frame,
            text="🔍 Найти",
            font=("Arial", 10, "bold"),
            bg="#2196F3",
            fg="white",
            width=12,
            command=self.search_users,
            cursor="hand2"
        ).pack(side=tk.LEFT, padx=5)
        
        tk.Button(
            button_frame,
            text="⭐ В избранное",
            font=("Arial", 10, "bold"),
            bg="#FF9800",
            fg="white",
            width=12,
            command=self.add_to_favourites,
            cursor="hand2"
        ).pack(side=tk.LEFT, padx=5)
        
        tk.Button(
            button_frame,
            text="📋 Показать избранное",
            font=("Arial", 10, "bold"),
            bg="#4CAF50",
            fg="white",
            width=15,
            command=self.show_favourites,
            cursor="hand2"
        ).pack(side=tk.LEFT, padx=5)
        
        # Список результатов
        list_frame = tk.Frame(self.root, bg="#2d2d2d")
        list_frame.pack(pady=10, fill=tk.BOTH, expand=True, padx=20)
        
        tk.Label(
            list_frame,
            text="📋 Результаты:",
            font=("Arial", 11, "bold"),
            bg="#2d2d2d",
            fg="white",
            anchor="w"
        ).pack(anchor="w")
        
        # Listbox с прокруткой
        scroll_frame = tk.Frame(list_frame)
        scroll_frame.pack(fill=tk.BOTH, expand=True)
        
        self.results_listbox = tk.Listbox(
            scroll_frame,
            height=12,
            font=("Arial", 10),
            bg="#3d3d3d",
            fg="white",
            selectbackground="#4CAF50",
            selectforeground="white",
            relief="solid",
            bd=1
        )
        self.results_listbox.pack(side=tk.LEFT, fill=tk.BOTH, expand=True)
        
        scrollbar = tk.Scrollbar(scroll_frame)
        scrollbar.pack(side=tk.RIGHT, fill=tk.Y)
        self.results_listbox.config(yscrollcommand=scrollbar.set)
        scrollbar.config(command=self.results_listbox.yview)
        
        # Кнопка удаления из избранного
        delete_button = tk.Button(
            self.root,
            text="❌ Удалить из избранного",
            font=("Arial", 10, "bold"),
            bg="#f44336",
            fg="white",
            width=20,
            command=self.remove_from_favourites,
            cursor="hand2"
        )
        delete_button.pack(pady=5)
        
        # Статусная строка
        self.status_label = tk.Label(
            self.root,
            text="Готов к поиску",
            font=("Arial", 9),
            bg="#2d2d2d",
            fg="#888888"
        )
        self.status_label.pack(pady=10)
    
    def search_users(self):
        """Поиск пользователей через GitHub API"""
        query = self.search_entry.get().strip()
        
        # Валидация
        if not query:
            messagebox.showwarning("Ошибка", "Введите имя пользователя для поиска!")
            return
        
        self.status_label.config(text="Поиск пользователей...")
        self.results_listbox.delete(0, tk.END)
        self.current_display_favourites = False
        
        try:
            # Запрос к GitHub API
            url = f"https://api.github.com/search/users?q={query}&per_page=20"
            response = requests.get(url)
            
            if response.status_code == 200:
                data = response.json()
                self.search_results = data.get("items", [])
                
                if self.search_results:
                    for user in self.search_results:
                        self.results_listbox.insert(tk.END, f"👤 {user['login']}")
                    self.status_label.config(
                        text=f"Найдено пользователей: {len(self.search_results)}"
                    )
                else:
                    self.status_label.config(text="Пользователи не найдены")
                    messagebox.showinfo("Результат", "Пользователи не найдены")
            else:
                self.status_label.config(text="Ошибка API")
                messagebox.showerror("Ошибка", "Не удалось выполнить поиск")
                
        except requests.exceptions.RequestException:
            self.status_label.config(text="Ошибка соединения")
            messagebox.showerror("Ошибка", "Проверьте подключение к интернету")
    
    def add_to_favourites(self):
        """Добавление выбранного пользователя в избранное"""
        if self.current_display_favourites:
            messagebox.showwarning("Ошибка", "Сначала выполните поиск пользователей")
            return
        
        selected = self.results_listbox.curselection()
        if not selected:
            messagebox.showwarning("Ошибка", "Выберите пользователя из списка!")
            return
        
        index = selected[0]
        user = self.search_results[index]
        
        # Проверяем, есть ли уже в избранном
        for fav in self.favourites:
            if fav["login"] == user["login"]:
                messagebox.showwarning("Ошибка", "Пользователь уже в избранном!")
                return
        
        # Добавляем в избранное
        self.favourites.append({
            "login": user["login"],
            "avatar_url": user["avatar_url"],
            "html_url": user["html_url"]
        })
        self.save_favourites()
        messagebox.showinfo("Успех", f"Пользователь {user['login']} добавлен в избранное!")
        self.status_label.config(text=f"Добавлен: {user['login']}")
    
    def show_favourites(self):
        """Отображение избранных пользователей"""
        self.results_listbox.delete(0, tk.END)
        
        if not self.favourites:
            self.results_listbox.insert(tk.END, "Список избранных пуст")
            self.status_label.config(text="Нет избранных пользователей")
        else:
            for user in self.favourites:
                self.results_listbox.insert(tk.END, f"⭐ {user['login']}")
            self.status_label.config(text=f"Избранных: {len(self.favourites)}")
        
        self.current_display_favourites = True
    
    def remove_from_favourites(self):
        """Удаление пользователя из избранного"""
        if not self.current_display_favourites:
            messagebox.showwarning("Ошибка", "Сначала откройте список избранного (📋 Показать избранное)")
            return
        
        selected = self.results_listbox.curselection()
        if not selected:
            messagebox.showwarning("Ошибка", "Выберите пользователя для удаления!")
            return
        
        index = selected[0]
        if index < len(self.favourites):
            removed = self.favourites.pop(index)
            self.save_favourites()
            self.show_favourites()
            messagebox.showinfo("Успех", f"Пользователь {removed['login']} удалён из избранного")
            self.status_label.config(text=f"Удалён: {removed['login']}")

# Запуск приложения
if __name__ == "__main__":
    root = tk.Tk()
    app = GitHubUserFinder(root)
    root.mainloop()
